def self.log(logpath, verbose, env, msg)
  if logpath
    system("mkdir #{logpath}") unless Dir.exist?("#{logpath}")
    open("#{logpath}/#{env}_error.log", 'a') do |f|
      f << "#{msg}\n"
    end
  end

  puts msg if verbose
end
